cmake_minimum_required(VERSION 3.22.1)

project(lzhphysics LANGUAGES C CXX VERSION 1.0.0)

# 指定动态库文件目录

# 指定可执行文件目录

# 指定源文件 
aux_source_directory(. SRCFILE)
aux_source_directory(${CMAKE_PROJECT_ROOT}/lzhapp SRCFILE)

# 设置编译时的链接库目录
link_directories(${CMAKE_LIBRARY_OUTPUT_DIRECTORY})

# 指定生成可执行程序
add_executable(lzhphysics ${SRCFILE})

# 生成位置无关代码
set_property(TARGET lzhphysics PROPERTY POSITION_INDEPENDENT_CODE ON)

# 设置 RPATH
set(CMAKE_BUILD_WITH_INSTALL_RPATH TRUE) 
 
#设置 RPATH路径
set_target_properties(${PROJECT_NAME} PROPERTIES INSTALL_RPATH "$ORIGIN/.")

# 指定头文件目录
include_directories(lzhphysics PRIVATE ${CMAKE_PROJECT_ROOT}/include)
include_directories(lzhphysics PRIVATE ${CMAKE_PROJECT_ROOT}/lzhapp)

# 指定依赖库
target_link_libraries(lzhphysics lzhcore)
